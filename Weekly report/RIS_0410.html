RIS in Ternay Alloys

1. Fractional concentration

\begin{equation*} X_{Fe}=\Omega C_{Fe} \end{equation*} \begin{equation*} X_{Cr}=\Omega C_{Cr} \end{equation*} \begin{equation*} X_{Ni}=\Omega C_{Ni} \end{equation*} Where
Average atomic volume \begin{equation*} \Omega = \frac{1}{N} \end{equation*} Atomic density \begin{equation*} N = 9.1\times 10^{28} \#/m^{3} \end{equation*}

2. Continuity equations

\begin{equation*} \frac{dX_{Fe}}{dt}=-\Omega \times \triangledown J_{Fe} \end{equation*} \begin{equation*} \frac{dX_{Cr}}{dt}=-\Omega \times \triangledown J_{Cr} \end{equation*} \begin{equation*} \frac{dX_{Ni}}{dt}=-\Omega \times \triangledown J_{Ni} \end{equation*} \begin{equation*} \frac{dX_{v}}{dt}=-\Omega \times \triangledown J_{v}+G_{v}-R\times X_{v}\times X_{i} \end{equation*} \begin{equation*} \frac{dX_{i}}{dt}=-\Omega \times \triangledown J_{i}+G_{i}-R\times X_{v}\times X_{i} \end{equation*}

3. Flux eqaution for atoms and defects

\begin{equation*} \Omega\times J_{Fe}=-\alpha D_{Fe}(1-X_{Fe})\triangledown X_{Fe}+[\alpha D_{Cr}\triangledown X_{Cr}+\alpha D_{Ni}\triangledown X_{Ni}+(d_{Fe}^{v}-D_{v})\triangledown X_{v}+(D_{i}-d_{Fe}^{i})\triangledown X_{i}]X_{Fe} \end{equation*} \begin{equation*} \Omega\times J_{Cr}=-\alpha D_{Cr}(1-X_{Cr})\triangledown X_{Cr}+[\alpha D_{Fe}\triangledown X_{Fe}+\alpha D_{Ni}\triangledown X_{Ni}+(d_{Cr}^{v}-D_{v})\triangledown X_{v}+(D_{i}-d_{Cr}^{i})\triangledown X_{i}]X_{Cr} \end{equation*} \begin{equation*} \Omega\times J_{Ni}=-\alpha D_{Ni}(1-X_{Ni})\triangledown X_{Ni}+[\alpha D_{Cr}\triangledown X_{Cr}+\alpha D_{Fe}\triangledown X_{Fe}+(d_{Ni}^{v}-D_{v})\triangledown X_{v}+(D_{i}-d_{Ni}^{i})\triangledown X_{i}]X_{Ni} \end{equation*} \begin{equation*} \Omega\times J_{v}=-D_{v}\triangledown X_{v}+\alpha X_{v}(d_{Fe}^{v}\triangledown X_{Fe}+d_{Cr}^{v}\triangledown X_{Cr}+d_{Ni}^{v}\triangledown X_{Ni}) \end{equation*} \begin{equation*} \Omega\times J_{i}=-D_{i}\triangledown X_{i}-\alpha X_{i}(d_{Fe}^{i}\triangledown X_{Fe}+d_{Cr}^{i}\triangledown X_{Cr}+d_{Ni}^{i}\triangledown X_{Ni}) \end{equation*}

4. Diffusion coefficients

\begin{equation*} D_{Fe}=d_{Fe}^{v}X_{v}+d_{Fe}^{i}X_{i} \end{equation*} \begin{equation*} D_{Cr}=d_{Cr}^{v}X_{v}+d_{Cr}^{i}X_{i} \end{equation*} \begin{equation*} D_{Ni}=d_{Ni}^{v}X_{v}+d_{Ni}^{i}X_{i} \end{equation*} \begin{equation*} D_{v}=d_{Fe}^{v}X_{Fe}+d_{Cr}^{v}X_{Cr}+d_{Ni}^{v}X_{Ni} \end{equation*} \begin{equation*} D_{i}=d_{Fe}^{i}X_{Fe}+d_{Cr}^{i}X_{Cr}+d_{Ni}^{i}X_{Ni} \end{equation*}

5. Diffusivities

Vacancies

\begin{equation*} \lambda_{a}^{v}=\lambda \end{equation*} \begin{equation*} d_{Fe}^{v}=\nu_{Fe}^{v}(\lambda_{a}^{v})^{2} \end{equation*} \begin{equation*} d_{Cr}^{v}=\nu_{Cr}^{v}(\lambda_{a}^{v})^{2} \end{equation*} \begin{equation*} d_{Ni}^{v}=\nu_{Ni}^{v}(\lambda_{a}^{v})^{2} \end{equation*}

Interstitial

\begin{equation*} \lambda_{a}^{i}=\sqrt{\frac{2}{3}}\lambda \end{equation*} \begin{equation*} d_{Fe}^{i}=\nu_{Fe}^{i}(\lambda_{a}^{i})^{2} \end{equation*} \begin{equation*} d_{Cr}^{i}=\nu_{Cr}^{i}(\lambda_{a}^{i})^{2} \end{equation*} \begin{equation*} d_{Ni}^{i}=\nu_{Ni}^{i}(\lambda_{a}^{i})^{2} \end{equation*}

6. Jump frequency

\begin{equation*} \nu_{Fe}^{v}=\nu_{Fe}^{v,0}e^{-\frac{E_{Fe}^{v,m}}{k_{b}T}} \end{equation*} \begin{equation*} \nu_{Cr}^{v}=\nu_{Cr}^{v,0}e^{-\frac{E_{Cr}^{v,m}}{k_{b}T}} \end{equation*} \begin{equation*} \nu_{Ni}^{v}=\nu_{Ni}^{v,0}e^{-\frac{E_{Ni}^{v,m}}{k_{b}T}} \end{equation*} \begin{equation*} \nu_{Fe}^{i}=f^{i}\nu^{i,0}e^{-\frac{E^{i,m}}{k_{b}T}} \end{equation*} \begin{equation*} \nu_{Cr}^{i}=f^{i}\nu^{i,0}e^{-\frac{E^{i,m}}{k_{b}T}} \end{equation*} \begin{equation*} \nu_{Ni}^{i}=f^{i}\nu^{i,0}e^{-\frac{E^{i,m}}{k_{b}T}} \end{equation*}

7. Migration energy

Pair interaction energy

Like atoms

\begin{equation*} E_{CrCr}=\frac{2E_{coh}^{Cr}}{Z} \end{equation*} \begin{equation*} E_{FeFe}=\frac{2E_{coh}^{Fe}}{Z} \end{equation*} \begin{equation*} E_{NiNi}=\frac{2E_{coh}^{Ni}}{Z} \end{equation*}

Unlike atoms

\begin{equation*} E_{NiCr}=\frac{E_{NiNi}+E_{CrCr}}{2}- E_{NiCr}^{ord}\end{equation*} \begin{equation*} E_{FeCr}=\frac{E_{FeFe}+E_{CrCr}}{2}- E_{FeCr}^{ord}\end{equation*} \begin{equation*} E_{FeNi}=\frac{E_{FeFe}+E_{NiNi}}{2}- E_{FeNi}^{ord}\end{equation*}

Atoms and vacancies

\begin{equation*} E_{Cr-v}=\frac{E_{coh}^{Cr}+E_{f}^{Cr-v}}{Z}\end{equation*} \begin{equation*} E_{Ni-v}=\frac{E_{coh}^{Ni}+E_{f}^{Ni-v}}{Z}\end{equation*} \begin{equation*} E_{Fe-v}=\frac{E_{coh}^{Fe}+E_{f}^{Fe-v}}{Z}\end{equation*}

Saddle point energy

\begin{equation*} ES_{pure}^{Cr}=E_{m}^{Cr-v}+Z(E_{CrCr}+E_{Cr-v})\end{equation*}

Vacancy migration energy

\begin{equation*} E_{Fe}^{v,m}=C_{Fe}(-\frac{3}{2}E_{Fe}^{coh}+\frac{1}{2}E_{Fe,pure}^{v,m}-\frac{1}{2}E_{Fe}^{v,f})+C_{Cr}(-E_{Fe}^{coh}-\frac{1}{2}E_{Cr}^{coh}+\frac{1}{2}E_{Cr,pure}^{v,m}-\frac{1}{2}E_{Cr}^{v,f}-ZE_{FeCr}^{ord})+C_{Ni}(-E_{Fe}^{coh}-\frac{1}{2}E_{Ni}^{coh}+\frac{1}{2}E_{Ni,pure}^{v,m}-\frac{1}{2}E_{Ni}^{v,f}-ZE_{NiFe}^{ord})+C_{v}(-E_{Fe}^{coh}-E_{Fe}^{v,f})+(\frac{3}{2}E_{Fe}^{coh}+\frac{1}{2}E_{Fe,pure}^{v,m}+\frac{1}{2}E_{Fe}^{v,f}) \end{equation*} \begin{equation*} E_{Cr}^{v,m}=C_{Cr}(-\frac{3}{2}E_{Cr}^{coh}+\frac{1}{2}E_{Cr,pure}^{v,m}-\frac{1}{2}E_{Cr}^{v,f})+C_{Fe}(-E_{Cr}^{coh}-\frac{1}{2}E_{Fe}^{coh}+\frac{1}{2}E_{Fe,pure}^{v,m}-\frac{1}{2}E_{Fe}^{v,f}-ZE_{FeCr}^{ord})+C_{Ni}(-E_{Cr}^{coh}-\frac{1}{2}E_{Ni}^{coh}+\frac{1}{2}E_{Ni,pure}^{v,m}-\frac{1}{2}E_{Ni}^{v,f}-ZE_{NiCr}^{ord})+C_{v}(-E_{Cr}^{coh}-E_{Cr}^{v,f})+(\frac{3}{2}E_{Cr}^{coh}+\frac{1}{2}E_{Cr,pure}^{v,m}+\frac{1}{2}E_{Cr}^{v,f}) \end{equation*} \begin{equation*} E_{Ni}^{v,m}=C_{Ni}(-\frac{3}{2}E_{Ni}^{coh}+\frac{1}{2}E_{Ni,pure}^{v,m}-\frac{1}{2}E_{Ni}^{v,f})+C_{Fe}(-E_{Ni}^{coh}-\frac{1}{2}E_{Fe}^{coh}+\frac{1}{2}E_{Fe,pure}^{v,m}-\frac{1}{2}E_{Fe}^{v,f}-ZE_{FeNi}^{ord})+C_{Cr}(-E_{Ni}^{coh}-\frac{1}{2}E_{Cr}^{coh}+\frac{1}{2}E_{Cr,pure}^{v,m}-\frac{1}{2}E_{Cr}^{v,f}-ZE_{NiCr}^{ord})+C_{v}(-E_{Ni}^{coh}-E_{Ni}^{v,f})+(\frac{3}{2}E_{Ni}^{coh}+\frac{1}{2}E_{Ni,pure}^{v,m}+\frac{1}{2}E_{Ni}^{v,f}) \end{equation*}

Interstitial migration energy

\begin{equation*} E^{i,m}=0.9eV \end{equation*}

7. Defect production rate

\begin{equation*} G_{v}=\eta\phi+\rho D_{v}(X_{v}^{th}-X_{v})\end{equation*} \begin{equation*} X_{v}^{th}=e^{\frac{S^{v,f}}{k_{b}}}e^{-\frac{E^{v,f}}{k_{b}T}} \end{equation*} \begin{equation*} \end{equation*} \begin{equation*} G_{i}=\eta\phi+\rho D_{i}(X_{i}^{th}-X_{i}) \end{equation*} \begin{equation*} X_{v}^{th}=small \end{equation*} \begin{equation*} R=[(\nu_{Fe}^{v}+\nu_{Fe}^{i})X_{Fe}+(\nu_{Cr}^{v}+\nu_{Cr}^{i})X_{Cr}+(\nu_{Ni}^{v}+\nu_{Ni}^{i})X_{Ni}]Z \end{equation*}

2) Set up all variables

Example of stiff problem: Sparse Brusselator System

The classic Brusselator system of equations is potentially large, stiff, and sparse. The Brusselator system models diffusion in a chemical reaction, and is represented by a system of equations $$u_{i}^{'} = 1 + u_{i}^{2}v_{i} - 4u_{i} + \alpha (N+1)^{2}(u_{i-1}-2u_{i}+u_{i+1})$$ $$v_{i}^{'} = 3u_{i} - u_{i}^{2}v_{i} + \alpha (N+1)^{2}(v_{i-1}-2v_{i}+v_{i+1}) $$ The function file brussode solves this set of equations on the time interval [0,10] with $\alpha = 1/50$. The initial conditions are $$u_{j}(0) = 1 + sin(2\pi\frac{j}{N+1})$$ $$v_{j}(0) = 3$$ A matlab code has a result

import numpy as np
import matplotlib.pyplot as plt
from matplotlib import cm
from scipy.integrate import solve_ivp
from mpl_toolkits.mplot3d import Axes3D

def Brusselator():
    N = 20
    t = [0, 10]
    y = np.zeros(2*N)
    y[:N] = 1 + np.sin(2*np.pi*np.arange(1, N+1)/(N+1))
    y[N:] = 3
    alpha = 1/50
    c = alpha*(N+1)**2

    def f(t, y):
        dydt = np.zeros(2*N)
        u = y[:N]
        v = y[N:]
        dydt[0] = 1 + v[0]*u[0]**2 - 4*u[0] + c*(1-2*u[0]+u[1])
        dydt[1:N-1] = 1 + v[1:-1]*u[1:-1]**2 - 4*u[1:-1] + c*(u[:-2]-2*u[1:-1]+u[2:])
        dydt[N-1] = 1 + v[-1]*u[-1]**2 - 4*u[-1] + c*(u[-2]-2*u[-1]+1)
        dydt[N] = 3*u[0] - v[0]*u[0]**2 + c*(3-2*v[0]+v[1])
        dydt[N+1:-1] = 3*u[1:-1] - v[1:-1]*u[1:-1]**2 + c*(v[:-2]-2*v[1:-1]+v[2:])
        dydt[-1] = 3*u[-1] - v[-1]*u[-1]**2 + c*(v[-2]-2*v[-1]+3)
        return dydt

    sol = solve_ivp(f, t, y, method="BDF", dense_output=True)

    time = np.arange(sol.y.shape[1])
    space = np.arange(N)
    x, y = np.meshgrid(time, space)
    z = sol.y[:N, ::-1]
    fig = plt.figure()
    ax = fig.gca(projection='3d')
    surf = ax.plot_surface(x, y, z, cmap=cm.coolwarm, linewidth=0, antialiased=False)
    fig.colorbar(surf, shrink=0.5, aspect=5)
    plt.show()

Solving Partial Differential Equations

A method of lines for partial differential equations (PDEs) where one reduces a PDE to a system of ODE and then applies standard methods. \begin{equation*} \frac{\partial u(x, t)}{\partial t}=\beta \frac{\partial^2u(x, t)} {\partial x^{2}}+f(x, t),x\in[0, L],t\in[0, T] \end{equation*} \begin{equation*} u(0, t)=s(t),t\in[0,T] \end{equation*} \begin{equation*} \frac{\partial}{\partial x}u(L,t)=0,t\in[0, T] \end{equation*} \begin{equation*} u(x,0)=I(x),x\in[0,L] \end{equation*} Discretizing the 2nd-order derivative in space with a finite difference on a mesh \begin{equation*} x_{i}=i\Delta x, i=1,...,N-1 \end{equation*} then ODE: \begin{equation*} \frac{\partial u_{i}(t)}{\partial t}=\beta \frac{u_{i+1}(t)-2u_{i}(t)+u_{i-1}(t)}{\Delta x^{2}}+f_{i}(t), i=1,...,N-1 \end{equation*} The boundary condition on x=0, u(0,t)=s(t), gives rise to the ODE \begin{equation*} u_{0}^{'}=s^{'}(t),u_{0}(0)=s(0) \end{equation*} At the other end, x=L, we use a centered difference approximation The boundary condition on x=0, u(0,t)=s(t), gives rise to the ODE \begin{equation*} \frac{u_{N+1}(t)-u_{N-1}(t)}{2\Delta x}=0 \end{equation*} and combine it with the scheme for i=N to obtain the modified boundary ODE \begin{equation*} \frac{\partial u_{N}(t)}{\partial t}=\beta \frac{2u_{N-1}(t)-2u_{N}(t)}{\partial x^{2}}+f_{N}(t) \end{equation*} To summarize, the ODE system reads \begin{equation*} \frac{du_{0}}{dt}=s^{'}(t) \end{equation*} \begin{equation*} \frac{du_{i}}{dt}=\frac{\beta}{\Delta x^{2}}(u_{i+1}(t)-2u_{i}(t)+2u_{i-1}(t))+f_{i}(t),i=1,...,N-1 \end{equation*} \begin{equation*} \frac{du_{N}}{dt}=\frac{2\beta}{\Delta x^{2}}(u_{N-1}(t)-u_{N}(t))+f_{i}(t) \end{equation*} The initial conditions are \begin{equation*} u_{0}(0)=s(0) \end{equation*} \begin{equation*} u_{i}(0)=I(x_{i}),i=1,...,N \end{equation*}

            _temp[:NSTEP] = CAFRAC[:NSTEP]
            CAFRAC = np.zeros(NSTEP)
            CAFRAC[:NSTEP] = _temp[:NSTEP]

            _temp[:NSTEP] = CBFRAC[:NSTEP]
            CBFRAC = np.zeros(NSTEP)
            CBFRAC[:NSTEP] = _temp[:NSTEP]

            _temp[:NSTEP] = CCFRAC[:NSTEP]
            CCFRAC = np.zeros(NSTEP)
            CCFRAC[:NSTEP] = _temp[:NSTEP]

            _temp[:NSTEP] = SFRAC[:NSTEP]
            SFRAC = np.zeros(NSTEP)
            SFRAC[:NSTEP] = _temp[:NSTEP]

            _temp[:NSTEP] = DFRAC[:NSTEP]
            DFRAC = np.zeros(NSTEP)
            DFRAC[:NSTEP] = _temp[:NSTEP]
        return TFRAC, CAFRAC, CBFRAC, CCFRAC, SFRAC, DFRAC

    def calculate_D(self):
        """
        This function is to calculate atom and defect diffusion coefficients.

        Formula:
            D_Fe = d_Fe_v * X_v + d_Fe_i * X_i
            D_Cr = d_Cr_v * X_v + d_Cr_i * X_i
            D_Ni = d_Ni_v * X_v + d_Ni_i * X_i
            D_v = d_Fe_v * X_Fe + d_Cr_v * X_Cr + d_Ni_v * X_Ni
            D_i = d_Fe_i * X_Fe + d_Cr_i * X_Cr + d_Ni_i * X_Ni
        where:
            X_Atom/Defect: fractional concentration of atom/defect
            d_Atom_Defect: diffusivity between atom and defect
            D_Atom/Defect: atom and defect diffusion coefficients
        """
        self.D_Fe = self.d_Fe_v * self.X_v + self.d_Fe_i * self.X_i

        self.D_Cr = self.d_Cr_v * self.X_v + self.d_Cr_i * self.X_i

        self.D_Ni = self.d_Ni_v * self.X_v + self.d_Ni_i * self.X_i

        self.D_v = self.d_Fe_v * self.X_Fe + self.d_Cr_v * self.X_Cr +\
                   self.d_Ni_v * self.X_Ni

        self.D_i = self.d_Fe_i * self.X_Fe + self.d_Cr_i * self.X_Cr +\
                   self.d_Ni_i * self.X_Ni
        #print("Finish diffusion coefficients")

    def calculate_d_Atom_Defect(self):
        """
        This function is to calculate diffusivity between atom and defect.
        
        Formula:
            LAMBDA_a_v = LAMBDA
            d_Fe_v = nu_Fe_v * LAMBDA_a_v^2
            d_Cr_v = nu_Cr_v * LAMBDA_a_v^2
            d_Ni_v = nu_Ni_v * LAMBDA_a_v^2
            
            LAMBDA_a_i = sqrt(2/3) * LAMBDA
            d_Fe_i = nu_Fe_i * LAMBDA_a_i^2
            d_Cr_i = nu_Cr_i * LAMBDA_a_i^2
            d_Ni_i = nu_Ni_i * LAMBDA_a_i^2
        where:
            LAMBDA: unit cell size
            LAMBDA_a_Defect: unit cell size between atom and defect
            nu_Atom_Defect: jump frequence between atom and defect
            d_Atom_Defect: diffusivity between defects and atoms
        """
        # Atom-vacancy jump distamce
        LAMBDA_a_v = LAMBDA
        self.d_Fe_v = self.nu_Fe_v * LAMBDA_a_v**2
        self.d_Cr_v = self.nu_Cr_v * LAMBDA_a_v**2
        self.d_Ni_v = self.nu_Ni_v * LAMBDA_a_v**2

        # Atom-interstitial jump distance
        LAMBDA_a_i = np.sqrt(2/3) * LAMBDA
        self.d_Fe_i = self.nu_Fe_i * LAMBDA_a_i**2
        self.d_Cr_i = self.nu_Cr_i * LAMBDA_a_i**2
        self.d_Ni_i = self.nu_Ni_i * LAMBDA_a_i**2

    def calculate_nu_Atome_Defect(self):
        """
        This function is to calculate jump frequence between atom and defect.
        Formula:
            nu_Fe_v = nu_Fe_v_0 * exp(-E_Fe_v_m/(kb*T))
            nu_Cr_v = nu_Cr_v_0 * exp(-E_Cr_v_m/(kb*T))
            nu_Ni_v = nu_Ni_v_0 * exp(-E_Ni_v_m/(kb*T))
            
            nu_Fe_i = f_i * nu_i_0 * exp(-E_i_m/(kb*T))
            nu_Cr_i = f_i * nu_i_0 * exp(-E_i_m/(kb*T))
            nu_Ni_i = f_i * nu_i_0 * exp(-E_i_m/(kb*T))
        where:
            nu_Fe_v_0: standard vacancy jump frequency for Fe
            E_Fe_v_m: vacancy migration energy for Fe
            kb: boltzmann constant
            T: temperature in kelvin
            nu_Fe_v: vacancy jump frequency for Fe
            
            f_i: interstitial correlation factor
            nu_i_0: standard interstitial jump frequency
            E_i_m: interstitial migration energy
            nu_Fe_i: interstitial jump frequency for Fe
        """
        kb = BOLTZ
        f_i = Jump_correlation_factor_i
        
        self.nu_Fe_v = self.nu_Fe_v_0 * np.exp(-self.E_Fe_v_m/(kb*self.T))
        self.nu_Cr_v = self.nu_Cr_v_0 * np.exp(-self.E_Cr_v_m/(kb*self.T))
        self.nu_Ni_v = self.nu_Ni_v_0 * np.exp(-self.E_Ni_v_m/(kb*self.T))
        
        self.nu_Fe_i = f_i * self.nu_i_0 * np.exp(-E_i_m/(kb*self.T))
        self.nu_Cr_i = f_i * self.nu_i_0 * np.exp(-E_i_m/(kb*self.T))
        self.nu_Ni_i = f_i * self.nu_i_0 * np.exp(-E_i_m/(kb*self.T))

    def calculate_E_Atom_v_m(self):
        """
        This function is to calculate jump frequence between atom and defect.
        
        Formula:
            E_Fe_v_m = C_Fe*(-3/2*E_Fe_coh+1/2*E_Fe_pure_v_m-1/2*E_Fe_v_f)+\
                C_Cr*(-E_Fe_coh-1/2*E_Cr_coh+1/2*E_Cr_pure_v_m-1/2*E_Cr_v_f-
                      Z*E_FeCr_ord)+\
                C_Ni*(-E_Fe_coh-1/2*E_Ni_coh+1/2*E_Ni_pure_v_m-1/2*E_Ni_v_f-
                      Z*E_NiFe_ord)+\
                C_v*(-E_Fe_coh-E_Fe_v_f)+\
                (3/2*E_Fe_coh+1/2*E_Fe_pure_v_m+1/2*E_Fe_v_f)
            E_Cr_v_m = C_Cr*(-3/2*E_Cr_coh+1/2*E_Cr_pure_v_m-1/2*E_Cr_v_f)+\
                C_Fe*(-E_Cr_coh-1/2*E_Fe_coh+1/2*E_Fe_pure_v_m-1/2*E_Fe_v_f-
                      Z*E_FeCr_ord)+\
                C_Ni*(-E_Cr_coh-1/2*E_Ni_coh+1/2*E_Ni_pure_v_m-1/2*E_Ni_v_f-
                      Z*E_NiCr_ord)+\
                C_v*(-E_Cr_coh-E_Cr_v_f)+\
                (3/2*E_Cr_coh+1/2*E_Cr_pure_v_m+1/2*E_Cr_v_f)
            E_Ni_v_m = C_Ni*(-3/2*E_Ni_coh+1/2*E_Ni_pure_v_m-1/2*E_Ni_v_f)+\
                C_Fe*(-E_Ni_coh-1/2*E_Fe_coh+1/2*E_Fe_pure_v_m-1/2*E_Fe_v_f-
                      Z*E_FeNi_ord)+\
                C_Cr*(-E_Ni_coh-1/2*E_Cr_coh+1/2*E_Cr_pure_v_m-1/2*E_Cr_v_f-
                      Z*E_NiCr_ord)+\
                C_v*(-E_Ni_coh-E_Ni_v_f)+\
                (3/2*E_Ni_coh+1/2*E_Ni_pure_v_m+1/2*E_Ni_v_f)
        where:
            C_Fe: concentration (= X_Fe*N, N is atomic density)
            E_Fe_coh: cohesive eergy for pure Fe
            E_Fe_pure_v_m: vacancy migration energy for pure Fe
            E_Fe_v_f: 
            E_FeCr_ord: ordering energy for Fe-Cr pairs
            Z: number of recombination volume
            E_Fe_v_m: vacancy migration energy for Fe
        """
        C_Fe = self.X_Fe
        C_Cr = self.X_Cr
        C_Ni = self.X_Ni
        C_v = self.X_v

        E_FeNi_ord = E_NiFe_ord

        self.E_Fe_v_m = C_Fe*(-3/2*E_Fe_coh+1/2*E_Fe_pure_v_m-1/2*E_Fe_v_f)+\
                        C_Cr*(-E_Fe_coh-1/2*E_Cr_coh+1/2*E_Cr_pure_v_m-
                              1/2*E_Cr_v_f-Z*E_FeCr_ord)+\
                        C_Ni*(-E_Fe_coh-1/2*E_Ni_coh+1/2*E_Ni_pure_v_m-
                              1/2*E_Ni_v_f-Z*E_NiFe_ord)+\
                        C_v*(-E_Fe_coh-E_Fe_v_f)+\
                        (3/2*E_Fe_coh+1/2*E_Fe_pure_v_m+1/2*E_Fe_v_f)

        self.E_Cr_v_m = C_Cr*(-3/2*E_Cr_coh+1/2*E_Cr_pure_v_m-1/2*E_Cr_v_f)+\
                        C_Fe*(-E_Cr_coh-1/2*E_Fe_coh+1/2*E_Fe_pure_v_m-
                              1/2*E_Fe_v_f-Z*E_FeCr_ord)+\
                        C_Ni*(-E_Cr_coh-1/2*E_Ni_coh+1/2*E_Ni_pure_v_m-
                              1/2*E_Ni_v_f-Z*E_NiCr_ord)+\
                        C_v*(-E_Cr_coh-E_Cr_v_f)+\
                        (3/2*E_Cr_coh+1/2*E_Cr_pure_v_m+1/2*E_Cr_v_f)

        self.E_Ni_v_m = C_Ni*(-3/2*E_Ni_coh+1/2*E_Ni_pure_v_m-1/2*E_Ni_v_f)+\
                        C_Fe*(-E_Ni_coh-1/2*E_Fe_coh+1/2*E_Fe_pure_v_m-
                              1/2*E_Fe_v_f-Z*E_FeNi_ord)+\
                        C_Cr*(-E_Ni_coh-1/2*E_Cr_coh+1/2*E_Cr_pure_v_m-
                              1/2*E_Cr_v_f-Z*E_NiCr_ord)+\
                        C_v*(-E_Ni_coh-E_Ni_v_f)+\
                        (3/2*E_Ni_coh+1/2*E_Ni_pure_v_m+1/2*E_Ni_v_f)
        
        EMA = E_Fe_pure_v_m
        EMB = E_Cr_pure_v_m
        EMC = E_Ni_pure_v_m
        
        ECOHA = E_Fe_coh
        ECOHB = E_Cr_coh
        ECOHC = E_Ni_coh

        EORDAB = E_FeCr_ord
        EORDAC = E_NiFe_ord
        EORDBC = E_NiCr_ord
        
        EAA = ECOHA/(Z/2)
        EBB = ECOHB/(Z/2)
        ECC = ECOHC/(Z/2)
        
        EFA = E_Fe_v_f
        EFB = E_Cr_v_f
        EFC = E_Ni_v_f
        
        EAV = (ECOHA+EFA)/Z
        EBV = (ECOHB+EFB)/Z
        ECV = (ECOHC+EFC)/Z
        
        ESA = EMA+Z*(EAA+EAV)
        ESB = EMB+Z*(EBB+EBV)
        ESC = EMC+Z*(ECC+ECV)
        
        EAB=0.5*(EAA+EBB)-EORDAB
        EAC=0.5*(EAA+ECC)-EORDAC
        EBC=0.5*(EBB+ECC)-EORDBC

        NA = self.X_Fe
        NB = self.X_Cr
        NC = self.X_Ni
        NV = self.X_v

        E_Fe_v_m = ((ESA+ESA*NA+ESB*NB+ESC*NC)/2)-\
                   ((Z*(NA*EAA+NB*EAB+NC*EAC+NV*EAV))+
                    (Z*(NA*EAV+NB*EBV+NC*ECV)))
        E_Cr_v_m = ((ESB+ESA*NA+ESB*NB+ESC*NC)/2)-\
                   ((Z*(NA*EAB+NB*EBB+NC*EBC+NV*EBV))+
                    (Z*(NA*EAV+NB*EBV+NC*ECV)))
        E_Ni_v_m = ((ESC+ESA*NA+ESB*NB+ESC*NC)/2)-\
                   ((Z*(NA*EAC+NB*EBC+NC*ECC+NV*ECV))+
                    (Z*(NA*EAV+NB*EBV+NC*ECV)))


    def calculate_defect_production_rates(self):
        """
        This function is to calculate defect production rates.

        Formula:
            G_v = eta*phi+rho*D_v*(X_v_th-X_v)
            X_v_th = exp(S_v_f/kb)*exp(-E_v_f/(kb*T))
            
            G_i = eta*phi+rho*D(X_i_th-X_i)
            X_i_th = small
            R = ((nu_Fe_v+nu_Fe_i)*X_Fe+
                 (nu_Cr_v+nu_Cr_i)*X_Cr+
                 (nu_Ni_v+nu_Ni_i)*X_Ni)*Z
        where:
            S_v_f: vacancy formation entropy (=kb)
            G_v: vacancy production rate
            E_v_f: vacancy formation energy (=1.9eV)
            G_i: interstitial production rate
            R: recombination production rate
        """
        self.G_v = eta*phi+rho*self.D_v*(self.X_v_th-self.X_v)
        self.G_i = eta*phi+rho*self.D_i*(self.X_i_th-self.X_i)
        self.R = ((self.nu_Fe_v+self.nu_Fe_i)*self.X_Fe+
                  (self.nu_Cr_v+self.nu_Cr_i)*self.X_Cr+
                  (self.nu_Ni_v+self.nu_Ni_i)*self.X_Ni)*Z

    def set_mesh(self):

    def div_X(self, X):
        """
        """
        dX = np.zeros(NSTEP)
        dX[:-1] = (X[1:]-X[:-1])/self.dx
        
        for i in range(1, NSTEP-1):
            f = (X[i-1]+X[i])/2
            b = (X[i]+X[i+1])/2
            dX[i] = (b-f)/self.dx
        '''
        dX[0] = dX[1]
        dX[-1] = dX[-2]
        '''
        return dX

    def div_OJ(self, OJ):
        """
        """
        OJ0 = 0
        dOJ = np.zeros(NSTEP)
        #
        #dOJ[1:-1] = (OJ[1:-1]-OJ[:-2])/self.mesh[1:-1]
        #dOJ[-1] = 2*(OJ[-1]-OJ[-2])/self.mesh[-1]
        dOJ[:-1] = (OJ[1:]-OJ[:-1])/self.dx
        for i in range(1, NSTEP-1):
            f = (OJ[i-1]+OJ[i])/2
            b = (OJ[i]+OJ[i+1])/2
            dOJ[i] = (b-f)/self.dx
        #dOJ[0] = 2*(OJ[0]-OJ0)/self.dx
        return dOJ

    def calculate_divergence(self):
        """
        Formula:
            d(X_Fe)/dt = -Omega*d(J_Fe)
            d(X_Cr)/dt = -Omega*d(J_Cr)
            d(X_Ni)/dt = -Omega*d(J_Ni)
            d(X_v)/dt = -Omega*d(J_v)+G_v-R*X_v*X_i
            d(X_i)/dt = -Omega*d(J_i)+G_i-R*X_v*X_i

            Omega*J_Fe = -alpha*D_Fe*(1-X_Fe)*d(X_Fe)+
                        [alpha*D_Cr*d(X_Cr)+alpha*D_Ni*d(X_Ni)+
                        (d_Fe_v-D_v)*d(X_v)+(D_i-d_Fe_i)*d(X_i)]*X_Fe
            Omega*J_Cr = -alpha*D_Cr*(1-X_Cr)*d(X_Cr)+
                        [alpha*D_Fe*d(X_Fe)+alpha*D_Ni*d(X_Ni)+
                        (d_Cr_v-D_v)*d(X_v)+(D_i-d_Cr_i)*d(X_i)]*X_Cr
            Omega*J_Ni = -alpha*D_Ni*(1-X_Ni)*d(X_Ni)+\
                        [alpha*D_Cr*d(X_Cr)+alpha*D_Fe*d(X_Fe)+
                        (d_Ni_v-D_v)*d(X_v)+(D_i-d_Ni_i)*d(X_i)]*X_Ni
            Omega*J_v = -D_v*d(X_v)+alpha*X_v*(d_Fe_v*d(X_Fe)+
                                               d_Cr_v*d(X_Cr)+
                                               d_Ni_v*d(X_Ni))
            Omega*J_i = -D_i*d(X_i)-alpha*X_i*(d_Fe_i*d(X_Fe)+
                                               d_Cr_i*d(X_Cr)+
                                               d_Ni_i*d(X_Ni))
        where:
            Omega: average atomic volume
            d(J_Fe): derivative of J_Fe
            alpha: thermodynamic factor
            D_Fe: diffusion coefficient for Fe
            X_Fe: fractional concentration for Fe
            J_Fe: Fe flux
        """
        self.div_X_Fe = self.div_X(self.X_Fe)
        self.div_X_Cr = self.div_X(self.X_Cr)
        self.div_X_Ni = self.div_X(self.X_Ni)
        self.div_X_v = self.div_X(self.X_v)
        self.div_X_i = self.div_X(self.X_i)

        self.OmegaJ_Fe = -alpha*self.D_Fe*(1-self.X_Fe)*self.div_X_Fe+\
                         (alpha*self.D_Cr*self.div_X_Cr+
                          alpha*self.D_Ni*self.div_X_Ni+
                          (self.d_Fe_v-self.D_v)*self.div_X_v+
                          (self.D_i-self.d_Fe_i)*self.div_X_i)*self.X_Fe

        self.OmegaJ_Cr = -alpha*self.D_Cr*(1-self.X_Cr)*self.div_X_Cr+\
                         (alpha*self.D_Fe*self.div_X_Fe+
                          alpha*self.D_Ni*self.div_X_Ni+
                          (self.d_Cr_v-self.D_v)*self.div_X_v+
                          (self.D_i-self.d_Cr_i)*self.div_X_i)*self.X_Cr
        

        self.OmegaJ_Ni = -alpha*self.D_Ni*(1-self.X_Ni)*self.div_X_Ni+\
                         (alpha*self.D_Cr*self.div_X_Cr+
                          alpha*self.D_Fe*self.div_X_Fe+
                          (self.d_Ni_v-self.D_v)*self.div_X_v+
                          (self.D_i-self.d_Ni_i)*self.div_X_i)*self.X_Ni

        self.OmegaJ_v = -self.D_v*self.div_X_v+\
                        alpha*self.X_v*(self.d_Fe_v*self.div_X_Fe+
                                        self.d_Cr_v*self.div_X_Cr+
                                        self.d_Ni_v*self.div_X_Ni)
        #self.OmegaJ_v[-1] = self.OmegaJ_v[-2]

        self.OmegaJ_i = -self.D_i*self.div_X_i-\
                        alpha*self.X_i*(self.d_Fe_i*self.div_X_Fe+
                                        self.d_Cr_i*self.div_X_Cr+
                                        self.d_Ni_i*self.div_X_Ni)
        #self.OmegaJ_i[-1] = self.OmegaJ_i[-2]

        self.div_OmegaJ_Fe = self.div_OJ(self.OmegaJ_Fe)
        self.div_OmegaJ_Cr = self.div_OJ(self.OmegaJ_Cr)
        self.div_OmegaJ_Ni = self.div_OJ(self.OmegaJ_Ni)
        self.div_OmegaJ_v = self.div_OJ(self.OmegaJ_v)
        self.div_OmegaJ_i = self.div_OJ(self.OmegaJ_i)

    def set_dudt(self, t, u):
        """
        Right-hand side of the ODEs
        """
        self.count += 1
        dudt = np.zeros(5*NSTEP)

        self.X_Fe = u[:NSTEP]
        self.X_Cr = u[NSTEP:2*NSTEP]
        self.X_Ni = u[2*NSTEP:3*NSTEP]
        self.X_v = u[3*NSTEP:4*NSTEP]
        self.X_i = u[4*NSTEP:]
        self.update_coefficients()

        if (self.E_Fe_v_m>1e10).sum():
            print(self.OmegaJ_Fe)
        #if 100<self.count<200:
            #print(self.OmegaJ_Fe)

        dudt[:NSTEP] = -self.div_OmegaJ_Fe
        dudt[NSTEP:2*NSTEP] = -self.div_OmegaJ_Cr
        dudt[2*NSTEP:3*NSTEP] = -self.div_OmegaJ_Ni
        dudt[3*NSTEP:4*NSTEP] = -self.div_OmegaJ_v+self.G_v-\
                                self.R*self.X_v*self.X_i
        dudt[4*NSTEP:] = -self.div_OmegaJ_i+self.G_i-self.R*self.X_v*self.X_i
        
        return dudt

    def solve(self):
        # Initial condition
        y = np.zeros(5*NSTEP)
        y[:NSTEP] = self.X_Fe
        y[NSTEP:2*NSTEP] = self.X_Cr
        y[2*NSTEP:3*NSTEP] = self.X_Ni
        y[3*NSTEP:4*NSTEP] = self.X_v
        y[4*NSTEP:] = self.X_i

        # Boundary condition
        y[3*NSTEP] = self.X_v_th
        y[4*NSTEP] = self.X_i_th
        t = [0, DOSE/DISPRT]

        sol = solve_ivp(self.set_dudt, t, y, method="BDF", dense_output=False,
                        vectorized=False)

        print(sol.message)
        print(sol.y.shape, self.count)

    def output(self):
        """
        """
        with open(OUTPUT_FILE, "a+") as f:
            f.write("")

    def update_coefficients(self):
        """
        Once fractional concentrations for Fe, Cr, Ni, v, i change, we have
        to update all coefficients.
        """
        self.calculate_E_Atom_v_m()
        self.calculate_nu_Atome_Defect()
        self.calculate_d_Atom_Defect()
        self.calculate_D()
        self.calculate_defect_production_rates()
        self.calculate_divergence()

    def plot(self):
        x = np.linspace(0, self.L, NSTEP)
        
        plt.figure(num=3, figsize=(8,5))
        plt.plot(x, self.X_Fe, color='red', linewidth=1.0, linestyle='--', label='Fe')
        plt.plot(x, self.X_Cr, color='green', linewidth=1.0, linestyle='--', label='Cr')
        plt.plot(x, self.X_Ni, color='blue', linewidth=1.0, linestyle='--', label='Ni')
        plt.legend()
        #plt.xlim((0, NSTEP))
        #plt.ylim((0, 1))
        plt.xlabel("Distance to grain boundary")
        plt.ylabel("Fractional concentration")
        plt.show()

    def main(self):
        """
        """
        self.empty_files()
        self.solve()
        self.plot()
